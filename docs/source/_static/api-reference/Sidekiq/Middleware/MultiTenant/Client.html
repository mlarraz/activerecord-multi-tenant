def call(_worker_class, msg, _queue, _redis_pool)
  if MultiTenant.current_tenant.present?
    msg['multi_tenant'] ||=
      {
        'class' => MultiTenant.current_tenant_class,
        'id' => MultiTenant.current_tenant_id
      }
  end

  yield
end
